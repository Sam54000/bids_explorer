bids_explorer.architecture

Main bids_explorer object.

This module contains the BidsArchitecture class, which represents the structure of a BIDS dataset. The class provides methods to query and interact with the dataset, allowing users to select specific files based on various criteria (subjects, sessions, datatypes, tasks, runs, etc.). It is designed for read-only operations and does not modify the underlying dataset structure.

272    def create_database(self) -> tuple[pd.DataFrame, pd.DataFrame]:
273        """Scan filesystem and build DataFrame of matching files.
274
275        Walks through the BIDS directory structure and creates two DataFrames:
276        one for valid files and one for errors encountered during scanning.
277
278        Returns:
279            tuple[pd.DataFrame, pd.DataFrame]: Tuple containing
280                                               (database, error_log)
281
282        Raises:
283            ValueError: If root directory is not set.
284        """
285        if not self.root:
286            raise ValueError("Root directory not set")
287
288        database_keys = [
289            "inode",
290            "root",
291            "subject",
292            "session",
293            "datatype",
294            "task",
295            "run",
296            "acquisition",
297            "description",
298            "suffix",
299            "extension",
300            "atime",
301            "mtime",
302            "ctime",
303            "filename",
304        ]
305
306        data: Dict[str, List[Any]] = {key: [] for key in database_keys}
307        error_flags: Dict[str, List[Any]] = {
308            "filename": [],
309            "error_type": [],
310            "error_message": [],
311            "inode": [],
312        }
313
314        for file in self._path_handler.generate():
315            if "test" in file.name.lower():
316                continue
317
318            try:
319                validate_bids_file(file)
320                bids_path = BidsPath.from_filename(file)
321                self._add_file_to_database(file, bids_path, data)
322            except Exception as e:
323                self._add_error_to_log(file, e, error_flags)
324
325        return self._create_dataframes(data, error_flags)

327    def _add_file_to_database(
328        self, file: Path, bids_path: BidsPath, data: Dict[str, List[Any]]
329    ) -> None:
330        """Add file information to database dictionary.
331
332        Args:
333            file: Path object representing the file.
334            bids_path: BidsPath instance containing parsed BIDS entities.

336        """
337        for key, value in bids_path.__dict__.items():

346        data["mtime"].append(int(file_stats.st_mtime))

403    def _get_range(
404        self,
405        dataframe_column: pd.core.series.Series,
406        start: int | str | None = None,
407        stop: int | str | None = None,
408    ) -> pd.core.series.Series:
409        if isinstance(start, str):
410            start = int(start)
411
412        if isinstance(stop, str):
413            stop = int(stop)
414
415        dataframe_column = dataframe_column.apply(lambda s: int(s))
416
417        if start is None or start == "*":
418            start = min(dataframe_column)
419
420        if stop is None or stop == "*":
421            stop = max(dataframe_column)
422
423        return (start <= dataframe_column) & (dataframe_column < stop)

440    def _interpret_string(
441        self,
442        dataframe_column: pd.core.series.Series,
443        string: str,
444    ) -> pd.core.series.Series:
445        """Interpret string patterns for database filtering.
446
447        Args:
448            dataframe_column: Series containing the column data to filter
449            string: String pattern to interpret (e.g. "1-5" for range)
450
451        Returns:
452            pd.core.series.Series: Boolean mask for matching rows
453        """
454        if "-" in string:
455            start, stop = string.split("-")
456            conditions = [
457                (start.isdigit() or start == "*"),
458                (stop.isdigit() or stop == "*"),
459            ]
460
461            if not all(conditions):
462                raise ValueError(
463                    "Input must be 2 digits separated by a `-` or "
464                    "1 digit and a wild card `*` separated by a `-`"
465                )
466
467            return self._get_range(
468                dataframe_column=dataframe_column,
469                start=int(start) if start.isdigit() else None,
470                stop=int(stop) if stop.isdigit() else None,
471            )
472
473        else:
474            return self._get_single_loc(dataframe_column, string)

484    def _create_mask(self, **kwargs: str | list[str] | None) -> pd.Series:
485        """Create boolean mask for filtering DataFrame.
486
487        Creates an optimized boolean mask for filtering the database based on
488        provided BIDS entities.
489
490        Args:
491            **kwargs: BIDS entities to filter by
492                      (e.g., subject='01', session='1').
493
494        Returns:
495            pd.Series: Boolean mask for filtering DataFrame.
496
497        Raises:
498            ValueError: If invalid selection keys are provided.
499        """
500        valid_keys = {
501            "subject",
502            "session",
503            "datatype",
504            "task",
505            "run",
506            "acquisition",
507            "description",
508            "suffix",
509            "extension",
510        }
511
512        invalid_keys = set(kwargs.keys()) - valid_keys
513        if invalid_keys:
514            raise ValueError(f"Invalid selection keys: {invalid_keys}")
515
516        valid_inodes = set(self._database.index)
517
518        for key, value in kwargs.items():
519            if value is None:
520                continue
521
522            if isinstance(value, list):
523                matching_inodes = set(
524                    self._database[self._database[key].isin(value)].index
525                )
526                valid_inodes &= matching_inodes
527                continue
528
529            if not isinstance(value, str):
530                continue
531
532            value = value.strip()
533            if not value:
534                continue
535
536            col = self._database[key]
537
538            if "-" in value and self._is_numerical(col):
539                start, stop = value.split("-")
540                conditions = [
541                    (start.isdigit() or start == "*"),
542                    (stop.isdigit() or stop == "*"),
543                ]
544
545                if not all(conditions):
546                    raise ValueError(
547                        "Input must be digits separated by a `-` or "
548                        "digits and a wild card `*` separated by a `-`"
549                    )
550
551                if all(conditions):
552                    col_numeric = pd.to_numeric(col, errors="coerce")
553
554                    start_val = (
555                        int(start) if start.isdigit() else col_numeric.min()
556                    )
557                    stop_val = (
558                        int(stop) if stop.isdigit() else col_numeric.max()
559                    )
560
561                    range_mask = (col_numeric >= start_val) & (
562                        col_numeric <= stop_val
563                    )
564                    matching_inodes = set(self._database[range_mask].index)
565                    valid_inodes &= matching_inodes
566
567            matching_inodes = set(self._database[col == value].index)
568            valid_inodes &= matching_inodes
569
570        return self._database.index.isin(valid_inodes)

592            >>> bids = BidsArchitecture(

607            ...         ]

620        mask = self._create_mask(**kwargs)
621        if inplace:

623            return self

626        if any(mask):
627            new_database = self._database.loc[mask]

629            new_database = pd.DataFrame()

631        setattr(new_instance, "_database", new_database)

272    def create_database(self) -> tuple[pd.DataFrame, pd.DataFrame]:
273        """Scan filesystem and build DataFrame of matching files.
274
275        Walks through the BIDS directory structure and creates two DataFrames:
276        one for valid files and one for errors encountered during scanning.
277
278        Returns:
279            tuple[pd.DataFrame, pd.DataFrame]: Tuple containing
280                                               (database, error_log)
281
282        Raises:
283            ValueError: If root directory is not set.
284        """
285        if not self.root:
286            raise ValueError("Root directory not set")
287
288        database_keys = [
289            "inode",
290            "root",
291            "subject",
292            "session",
293            "datatype",
294            "task",
295            "run",
296            "acquisition",
297            "description",
298            "suffix",
299            "extension",
300            "atime",
301            "mtime",
302            "ctime",
303            "filename",
304        ]
305
306        data: Dict[str, List[Any]] = {key: [] for key in database_keys}
307        error_flags: Dict[str, List[Any]] = {
308            "filename": [],
309            "error_type": [],
310            "error_message": [],
311            "inode": [],
312        }
313
314        for file in self._path_handler.generate():
315            if "test" in file.name.lower():
316                continue
317
318            try:
319                validate_bids_file(file)
320                bids_path = BidsPath.from_filename(file)
321                self._add_file_to_database(file, bids_path, data)
322            except Exception as e:
323                self._add_error_to_log(file, e, error_flags)
324
325        return self._create_dataframes(data, error_flags)
